apiVersion: argoproj.io/v1alpha1
kind: Application
metadata:
  name: mysql
  namespace: argocd
spec:
  project: default

  source:
    repoURL: https://github.com/devpandai/mysql-deploy.git
    targetRevision: main
    path: .

  destination:
    server: https://kubernetes.default.svc
    namespace: cicd

  syncPolicy:
    automated:        # Auto sync changes from repo
      prune: true     # Hapus resource yang tidak ada di repo
      selfHeal: true  # Perbaiki jika ada perubahan manual di cluster
    syncOptions:
      - CreateNamespace=true  # Buat namespace cicd kalau belum ada
